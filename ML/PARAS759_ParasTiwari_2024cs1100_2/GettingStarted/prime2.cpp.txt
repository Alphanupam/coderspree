

#include <iostream>
int prime(int n)
{int j=2;
    for(int j=2;j<n;j++)
        {
            if(n%j==0)
            {
                return 0;
                
            }
            
        }
        if(j==n)
        {
            return n;
        }
        
}
using namespace std;
int main(int argc, char **argv){
    int low,a, high;
    cin >> low >> high;

    for(int i=low;i<high;i++)
    {
        a=prime(i);
        cout<<a<<"\n";
    }

}
